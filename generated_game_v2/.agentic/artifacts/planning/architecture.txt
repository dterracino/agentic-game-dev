# Qix Clone — Architecture Plan (Pygame renderer)

## 0. Requirement conflict resolution

The brief says "ModernGL rendering" but the explicit **Requested renderer: pygame** field overrides prose in the brief. Per the standing instruction ("ModernGL only when requested"), the explicit field is the actual request, so this plan uses **pygame's 2D `Surface`/`draw` API only**. No ModernGL, no GPU shader pipeline. If GPU-accelerated rendering is truly wanted, restate it explicitly as the requested renderer.

---

## 1. Dependencies

| PyPI package | Import name | Version constraint | Reason |
|---|---|---|---|
| `pygame-ce` | `pygame` | `>=2.5,<3.0` | Window/surface creation, event/input polling, 2D primitive drawing, clock/dt timing, audio. Actively maintained fork of `pygame`. |

Everything else uses the standard library: `dataclasses`, `enum`, `math`, `random`, `typing`, `array`, `functools`, `pathlib`. No numpy — the play-grid is small enough for plain lists/`array`.

## 3. Import direction (acyclic)

```
core          -> (nothing in game/)
utils         -> core
entities      -> core, utils
world         -> core, utils, entities
engine        -> core, utils            (never imports states/entities/world)
states        -> core, utils, world, entities, engine.renderer/input (types only)
ui            -> core, utils
main.py       -> engine, states, core
```

Rule enforced: `engine/app.py` never imports from `states/`. Instead `main.py` builds the `StateManager` with concrete states and hands it to `App`. `states/*` receive engine services (`Renderer`, `InputState`, `AssetManager`) only as constructor-injected objects typed via `core/context.py`, never by importing `engine.app`.

## 4. Core shared types

```python
# core/constants.py
SCREEN_W, SCREEN_H = 1024, 768
FPS_CAP = 240
CELL_SIZE = 4                 # px per grid cell
GRID_W, GRID_H = SCREEN_W // CELL_SIZE, SCREEN_H // CELL_SIZE

PLAYER_DRAW_SPEED = 90.0       # px/s while cutting
QIX_SPEED = 160.0
SPARX_SPEED = 130.0
CLAIM_TARGET_PCT = 0.75
STARTING_LIVES = 3

# core/types.py
from dataclasses import dataclass
from enum import Enum, auto

@dataclass(frozen=True, slots=True)
class Vec2:
    x: float
    y: float
    def __add__(self, o: "Vec2") -> "Vec2": ...
    def __mul__(self, s: float) -> "Vec2": ...
    def length(self) -> float: ...
    def normalized(self) -> "Vec2": ...

class CellState(Enum):
    EMPTY = 0
    CLAIMED = 1
    BORDER = 2
    TRAIL = 3

class GameState(Enum):
    MENU = auto()
    PLAYING = auto()
    PAUSED = auto()
    GAME_OVER = auto()
    VICTORY = auto()

@dataclass(frozen=True, slots=True)
class InputIntent:
    move: Vec2                # normalized-ish direction, (0,0) if idle
    draw_held: bool
    pause_pressed: bool
    confirm_pressed: bool
    quit_requested: bool

# core/context.py
from dataclasses import dataclass
import pygame
from game.engine.renderer import Renderer
from game.engine.assets import AssetManager

@dataclass(slots=True)
class GameContext:
    screen: pygame.Surface
    renderer: Renderer
    assets: AssetManager
    dt: float = 0.0
    score: int = 0
    lives: int = STARTING_LIVES
    level: int = 1
```

`GameContext` is the single object threaded through states/entities every frame — nothing reaches back into `App`.

## 5. Engine layer

```python
# engine/input.py
class InputHandler:
    def __init__(self) -> None: ...
    def poll(self, events: list[pygame.event.Event]) -> InputIntent: ...

# engine/assets.py
class AssetManager:
    def __init__(self, base_path: Path) -> None: ...
    def get_font(self, name: str, size: int) -> pygame.font.Font: ...
    def get_sound(self, name: str) -> pygame.mixer.Sound | None: ...

# engine/renderer.py
class Renderer:
    """Thin wrapper over pygame.draw + surface blits; keeps drawing
    calls out of entity/state logic bodies where practical."""
    def __init__(self, surface: pygame.Surface) -> None: ...
    def clear(self, color: tuple[int,int,int]) -> None: ...
    def line(self, p1: Vec2, p2: Vec2, color, width: int = 1) -> None: ...
    def polyline(self, points: list[Vec2], color, width: int = 1) -> None: ...
    def rect(self, rect: pygame.Rect, color, width: int = 0) -> None: ...
    def circle(self, center: Vec2, radius: float, color, width: int = 0) -> None: ...
    def text(self, text: str, pos: Vec2, font: pygame.font.Font, color) -> None: ...
    def present(self) -> None: ...   # pygame.display.flip()

# engine/app.py
class App:
    def __init__(self, screen: pygame.Surface, clock: pygame.time.Clock,
                 state_manager: "StateManager", input_handler: InputHandler) -> None: ...
    def run(self) -> None:
        """Owns the loop: poll events -> input -> dt -> state.update/draw -> present."""
```

`App.run()`:
```python
def run(self) -> None:
    running = True
    while running:
        dt = self.clock.tick(FPS_CAP) / 1000.0
        events = pygame.event.get()
        intent = self.input_handler.poll(events)
        if intent.quit_requested:
            running = False
            continue
        self.state_manager.handle_input(intent)
        self.state_manager.update(dt)
        self.state_manager.draw()
        pygame.display.flip()
    pygame.quit()
```

## 6. State machine

```python
# states/base_state.py
class State(ABC):
    def __init__(self, ctx: GameContext, manager: "StateManager") -> None: ...
    def on_enter(self) -> None: ...
    def on_exit(self) -> None: ...
    @abstractmethod
    def handle_input(self, intent: InputIntent) -> None: ...
    @abstractmethod
    def update(self, dt: float) -> None: ...
    @abstractmethod
    def draw(self) -> None: ...

# states/state_manager.py
class StateManager:
    def __init__(self, ctx: GameContext) -> None: ...
    def register(self, state_id: GameState, state: State) -> None: ...
    def switch_to(self, state_id: GameState) -> None: ...
    def handle_input(self, intent: InputIntent) -> None: ...  # delegates to current
    def update(self, dt: float) -> None: ...
    def draw(self) -> None: ...
```

Transitions (explicit, no implicit fallthrough):

```
MENU        --confirm-->            PLAYING
PLAYING     --pause-->              PAUSED
PAUSED      --pause/confirm-->      PLAYING
PLAYING     --lives==0-->           GAME_OVER
PLAYING     --claimed>=75%-->       VICTORY
GAME_OVER   --confirm-->            MENU
VICTORY     --confirm-->            MENU
```

Each concrete state (`PlayingState`, etc.) requests a transition by calling `self.manager.switch_to(...)`; it never mutates another state directly.

```python
def flood_fill_region(grid: Grid, seed: tuple[int,int],
                       blocked: frozenset[CellState]) -> set[tuple[int,int]]:

def resolve_cut(grid: Grid, cut_cells: list[tuple[int,int]],
                 qix_cell: tuple[int,int]) -> int:
    """Marks cut_cells as BORDER, splits the enclosed empty area into

class Trail:
    def __init__(self) -> None: ...
```

## 8. Entities

```python
# entities/base.py
class Entity(ABC):
    position: Vec2
    @abstractmethod
    def update(self, dt: float, board: Board) -> None: ...
    @abstractmethod
    def draw(self, renderer: Renderer) -> None: ...

# entities/player.py
class Player(Entity):
    def __init__(self, start: Vec2) -> None: ...
    def apply_intent(self, intent: InputIntent, board: Board) -> None: ...
    def update(self, dt: float, board: Board) -> None: ...
    def draw(self, renderer: Renderer) -> None: ...
    def is_drawing(self) -> bool: ...
    def check_self_death(self, board: Board) -> bool: ...   # trail hit by own re-entry error

# entities/qix.py
class Qix(Entity):
    def __init__(self, start: Vec2) -> None: ...
    def update(self, dt: float, board: Board) -> None: ...   # bounces within empty region
    def draw(self, renderer: Renderer) -> None: ...
    def collides_with_point(self, p: Vec2, radius: float) -> bool: ...

# entities/sparx.py
class Sparx(Entity):
    def __init__(self, start_cell: tuple[int,int], speed: float) -> None: ...
    def update(self, dt: float, board: Board) -> None: ...   # travels border_polyline
    def draw(self, renderer: Renderer) -> None: ...
    def collides_with_point(self, p: Vec2, radius: float) -> bool: ...
```

Movement is always `position = position + direction * speed * dt`, so all speeds are px/s and frame-rate independent.

## 9. Playing state (ties it together)

```python
# states/playing_state.py
class PlayingState(State):
    def __init__(self, ctx: GameContext, manager: StateManager) -> None:
        self.board = Board()
        self.player = Player(start=...)
        self.qix = Qix(start=...)
        self.sparx: list[Sparx] = [...]

    def handle_input(self, intent: InputIntent) -> None:
        if intent.pause_pressed:
            self.manager.switch_to(GameState.PAUSED)
            return
        self.player.apply_intent(intent, self.board)

    def update(self, dt: float) -> None:
        self.player.update(dt, self.board)
        self.qix.update(dt, self.board)
        for s in self.sparx:
            s.update(dt, self.board)
        self._check_collisions()
        if self.board.percent_claimed() >= CLAIM_TARGET_PCT:
            self.manager.switch_to(GameState.VICTORY)
        elif self.ctx.lives <= 0:
            self.manager.switch_to(GameState.GAME_OVER)

    def draw(self) -> None:
        r = self.ctx.renderer
        r.clear((0, 0, 0))
        r.polyline(self.board.border_polyline(), (0, 200, 255), 2)
        r.polyline(self.player.trail_points, (255, 255, 0), 2) if self.player.is_drawing() else None
        self.player.draw(r); self.qix.draw(r)
        for s in self.sparx: s.draw(r)
        hud.draw(r, self.ctx)

    def _check_collisions(self) -> None:
        """Qix/Sparx vs player -> lose life, reset player, clear trail.
           Qix vs trail segment -> lose life."""
```
